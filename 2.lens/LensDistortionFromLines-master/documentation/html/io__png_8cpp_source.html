131 static void *io_png_read_raw(const char *fname,
132  size_t * nxp, size_t * nyp, size_t * ncp,
133  int png_transform, int dtype) {
134  png_byte png_sig[PNG_SIG_LEN];
135  png_structp png_ptr;
136  png_infop info_ptr;
137  png_bytepp row_pointers;
138  png_bytep row_ptr;
139  /* volatile: because of setjmp/longjmp */
140  FILE *volatile fp = NULL;
141  void *data = NULL;
142  unsigned char *data_u8 = NULL;
143  unsigned char *data_u8_ptr = NULL;
144  float *data_f32 = NULL;
145  float *data_f32_ptr = NULL;
146  size_t size;
147  size_t i, j, k;
148  /* local error structure */
149  _io_png_err_t err;
150 
151  /* parameters check */
152  if (NULL == fname || NULL == nxp || NULL == nyp || NULL == ncp)
153  return NULL;
154  if (IO_PNG_U8 != dtype && IO_PNG_F32 != dtype)
155  return NULL;
156 
157  /* open the PNG input file */
158  if (0 == strcmp(fname, "-"))
159  fp = stdin;
160  else if (NULL == (fp = fopen(fname, "rb")))
161  return NULL;
162 
163  /* read in some of the signature bytes and check this signature */
164  if ((PNG_SIG_LEN != fread(png_sig, 1, PNG_SIG_LEN, fp))
165  || 0 != png_sig_cmp(png_sig, (png_size_t) 0, PNG_SIG_LEN))
166  return _io_png_read_abort(fp, NULL, NULL);
167 
168  /*
169  * create and initialize the png_struct
170  * with local error handling
171  */
172  if (NULL == (png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING,
173  &err, &_io_png_err_hdl,
174  NULL)))
175  return _io_png_read_abort(fp, NULL, NULL);
176 
177  /* allocate/initialize the memory for image information */
178  if (NULL == (info_ptr = png_create_info_struct(png_ptr)))
179  return _io_png_read_abort(fp, &png_ptr, NULL);
180 
181  /* handle read errors */
182  if (setjmp(err.jmpbuf))
183  /* if we get here, we had a problem reading from the file */
184  return _io_png_read_abort(fp, &png_ptr, NULL);
185 
186  /* set up the input control using standard C streams */
187  png_init_io(png_ptr, fp);
188 
189  /* let libpng know that some bytes have been read */
190  png_set_sig_bytes(png_ptr, PNG_SIG_LEN);
191 
192  /*
193  * set the read filter transforms, to get 8bit RGB whatever the
194  * original file may contain:
195  * PNG_TRANSFORM_STRIP_16 strip 16-bit samples to 8 bits
196  * PNG_TRANSFORM_PACKING expand 1, 2 and 4-bit
197  * samples to bytes
198  */
199  png_transform |= (PNG_TRANSFORM_STRIP_16 | PNG_TRANSFORM_PACKING);
200 
201  /* read in the entire image at once */
202  png_read_png(png_ptr, info_ptr, png_transform, NULL);
203 
204  /* get image informations */
205  *nxp = (size_t) png_get_image_width(png_ptr, info_ptr);
206  *nyp = (size_t) png_get_image_height(png_ptr, info_ptr);
207  *ncp = (size_t) png_get_channels(png_ptr, info_ptr);
208  row_pointers = png_get_rows(png_ptr, info_ptr);
209 
210  /*
211  * allocate the output data RGB array
212  * de-interlace and convert png RGB RGB RGB 8bit to RRR GGG BBB
213  * the image is de-interlaced layer after layer
214  * this generic loop also works for one single channel
215  */
216  size = *nxp * *nyp * *ncp;
217  switch (dtype) {
218  case IO_PNG_U8:
219  if (NULL == (data_u8 =
220  (unsigned char *) malloc(size * sizeof(unsigned char))))
221  return _io_png_read_abort(fp, &png_ptr, &info_ptr);
222  data = (void *) data_u8;
223  for (k = 0; k < *ncp; k++) {
224  /* channel loop */
225  data_u8_ptr = data_u8 + (size_t) (*nxp * *nyp * k);
226  for (j = 0; j < *nyp; j++) {
227  /* row loop */
228  row_ptr = row_pointers[j] + k;
229  for (i = 0; i < *nxp; i++) {
230  /* pixel loop */
231  *data_u8_ptr++ = (unsigned char) *row_ptr;
232  row_ptr += *ncp;
233  }
234  }
235  }
236  break;
237  case IO_PNG_F32:
238  if (NULL == (data_f32 = (float *) malloc(size * sizeof(float))))
239  return _io_png_read_abort(fp, &png_ptr, &info_ptr);
240  data = (void *) data_f32;
241  for (k = 0; k < *ncp; k++) {
242  /* channel loop */
243  data_f32_ptr = data_f32 + (size_t) (*nxp * *nyp * k);
244  for (j = 0; j < *nyp; j++) {
245  /* row loop */
246  row_ptr = row_pointers[j] + k;
247  for (i = 0; i < *nxp; i++) {
248  /* pixel loop */
249  *data_f32_ptr++ = (float) *row_ptr;
250  row_ptr += *ncp;
251  }
252  }
253  }
254  break;
255  }
256 
257  /* clean up and free any memory allocated, close the file */
258  (void) _io_png_read_abort(fp, &png_ptr, &info_ptr);
259 
260  return data;
261 }

333 unsigned char *io_png_read_u8_gray(const char *fname,
334  size_t * nxp, size_t * nyp) {
335  size_t nc;
336  unsigned char *img;
337 
338  /* read the image */
339  img = (unsigned char *) io_png_read_raw(fname, nxp, nyp, &nc,
340  PNG_TRANSFORM_STRIP_ALPHA,
341  IO_PNG_U8);
342  if (NULL == img)
343  /* error */
344  return NULL;
345  if (1 == nc)
346  /* already gray */
347  return img;
348  else {
349  /* convert to gray */
350  size_t i, size;
351  unsigned char *img_r, *img_g, *img_b;
352 
353  /*
354  * RGB->gray conversion
355  * Y = (6968 * R + 23434 * G + 2366 * B) / 32768
356  * integer approximation of
357  * Y = Cr* R + Cg * G + Cb * B
358  * with
359  * Cr = 0.212639005871510
360  * Cg = 0.715168678767756
361  * Cb = 0.072192315360734
362  * derived from ITU BT.709-5 (Rec 709) sRGB and D65 definitions
363  * http://www.itu.int/rec/R-REC-BT.709/en
364  */
365  size = *nxp * *nyp;
366  img_r = img;
367  img_g = img + size;
368  img_b = img + 2 * size;
369  for (i = 0; i < size; i++)
370  /*
371  * if int type is less than 24 bits, we use long ints,
372  * guaranteed to be >=32 bit
373  */
374 #if (UINT_MAX>>24 == 0)
375 #define CR 6968ul
376 #define CG 23434ul
377 #define CB 2366ul
378 #else
379 #define CR 6968u
380 #define CG 23434u
381 #define CB 2366u
382 #endif
383  /* (1 << 14) is added for rounding instead of truncation */
384  img[i] = (unsigned char) ((CR * img_r[i] + CG * img_g[i]
385  + CB * img_b[i] + (1 << 14)) >> 15);
386 #undef CR
387 #undef CG
388 #undef CB
389  /* resize and return the image */
390  img = (unsigned char *) realloc(img, size * sizeof(unsigned char));
391  return img;
392  }
393 }

545 static int io_png_write_raw(const char *fname, const void *data,
546  size_t nx, size_t ny, size_t nc, int dtype) {
547  png_structp png_ptr;
548  png_infop info_ptr;
549  png_byte *idata = NULL, *idata_ptr = NULL;
550  png_bytep *row_pointers = NULL;
551  png_byte bit_depth;
552  /* volatile: because of setjmp/longjmp */
553  FILE *volatile fp;
554  const unsigned char *data_u8 = NULL;
555  const unsigned char *data_u8_ptr = NULL;
556  const float *data_f32 = NULL;
557  const float *data_f32_ptr = NULL;
558  float tmp;
559  int color_type, interlace, compression, filter;
560  size_t size;
561  size_t i, j, k;
562  /* error structure */
563  _io_png_err_t err;
564 
565  /* parameters check */
566  if (0 >= nx || 0 >= ny || 0 >= nc)
567  return -1;
568  if (NULL == fname || NULL == data)
569  return -1;
570  if (IO_PNG_U8 != dtype && IO_PNG_F32 != dtype)
571  return -1;
572 
573  /* open the PNG output file */
574  if (0 == strcmp(fname, "-"))
575  fp = stdout;
576  else if (NULL == (fp = fopen(fname, "wb")))
577  return -1;
578 
579  /* allocate the interlaced array and row pointers */
580  size = nx * ny * nc;
581  if (NULL == (idata = (png_byte *) malloc(size * sizeof(png_byte))))
582  return _io_png_write_abort(fp, NULL, NULL, NULL, NULL);
583 
584  if (NULL == (row_pointers = (png_bytep *) malloc(ny * sizeof(png_bytep))))
585  return _io_png_write_abort(fp, idata, NULL, NULL, NULL);
586 
587  /*
588  * create and initialize the png_struct
589  * with local error handling
590  */
591  if (NULL == (png_ptr = png_create_write_struct(PNG_LIBPNG_VER_STRING,
592  &err, &_io_png_err_hdl,
593  NULL)))
594  return _io_png_write_abort(fp, idata, row_pointers, NULL, NULL);
595 
596  /* allocate/initialize the memory for image information */
597  if (NULL == (info_ptr = png_create_info_struct(png_ptr)))
598  return _io_png_write_abort(fp, idata, row_pointers, &png_ptr, NULL);
599 
600  /* handle write errors */
601  if (0 != setjmp(err.jmpbuf))
602  /* if we get here, we had a problem writing to the file */
603  return _io_png_write_abort(fp, idata, row_pointers, &png_ptr,
604  &info_ptr);
605 
606  /* set up the input control using standard C streams */
607  png_init_io(png_ptr, fp);
608 
609  /* set image informations */
610  bit_depth = 8;
611  switch (nc) {
612  case 1:
613  color_type = PNG_COLOR_TYPE_GRAY;
614  break;
615  case 2:
616  color_type = PNG_COLOR_TYPE_GRAY_ALPHA;
617  break;
618  case 3:
619  color_type = PNG_COLOR_TYPE_RGB;
620  break;
621  case 4:
622  color_type = PNG_COLOR_TYPE_RGB_ALPHA;
623  break;
624  default:
625  png_destroy_read_struct(&png_ptr, NULL, NULL);
626  free(row_pointers);
627  free(idata);
628  (void) fclose(fp);
629  return -1;
630  }
631  interlace = PNG_INTERLACE_ADAM7;
632  compression = PNG_COMPRESSION_TYPE_BASE;
633  filter = PNG_FILTER_TYPE_BASE;
634 
635  /* set image header */
636  png_set_IHDR(png_ptr, info_ptr, (png_uint_32) nx, (png_uint_32) ny,
637  bit_depth, color_type, interlace, compression, filter);
638  /* TODO : significant bit (sBIT), gamma (gAMA), comments (text) chunks */
639  png_write_info(png_ptr, info_ptr);
640 
641  /*
642  * interlace and convert RRR GGG BBB to RGB RGB RGB
643  * the image is interlaced layer after layer
644  * this involves more memory exchange, but allows a generic loop
645  */
646  switch (dtype) {
647  case IO_PNG_U8:
648  data_u8 = (unsigned char *) data;
649  for (k = 0; k < nc; k++) {
650  /* channel loop */
651  data_u8_ptr = data_u8 + (size_t) (nx * ny * k);
652  idata_ptr = idata + (size_t) k;
653  for (j = 0; j < ny; j++) {
654  /* row loop */
655  for (i = 0; i < nx; i++) {
656  /* pixel loop */
657  *idata_ptr = (png_byte) * data_u8_ptr++;
658  idata_ptr += nc;
659  }
660  }
661  }
662  break;
663  case IO_PNG_F32:
664  data_f32 = (float *) data;
665  for (k = 0; k < nc; k++) {
666  /* channel loop */
667  data_f32_ptr = data_f32 + (size_t) (nx * ny * k);
668  idata_ptr = idata + (size_t) k;
669  for (j = 0; j < ny; j++) {
670  /* row loop */
671  for (i = 0; i < nx; i++) {
672  /* pixel loop */
673  tmp = floor(*data_f32_ptr++ + .5);
674  *idata_ptr = (png_byte) (tmp < 0. ? 0. :
675  (tmp > 255. ? 255. : tmp));
676  idata_ptr += nc;
677  }
678  }
679  }
680  break;
681  }
682 
683  /* set row pointers */
684  for (j = 0; j < ny; j++)
685  row_pointers[j] = idata + (size_t) (nc * nx * j);
686 
687  /* write out the entire image and end it */
688  png_write_image(png_ptr, row_pointers);
689  png_write_end(png_ptr, info_ptr);
690 
691  /* clean up and free any memory allocated, close the file */
692  (void) _io_png_write_abort(fp, idata, row_pointers, &png_ptr, &info_ptr);
693 
694  return 0;
695 }
