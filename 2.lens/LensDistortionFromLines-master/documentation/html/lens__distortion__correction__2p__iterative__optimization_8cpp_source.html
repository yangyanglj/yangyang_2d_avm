32 double energy_minimization(lens_distortion_model &ldm, image_primitives &ip,
33  int w, int h, bool opt_center)

39  lens_distortion_model tf_ldm, tt_ldm;
40  if(!opt_center)
41  {

43  {
44  error = model_center_estimation_2p_polynomial(ldm.get_distortion_center(),
45  ip.get_lines(), ldm, w, h,

47  }

50  error = model_center_estimation_2p_quotient(ldm.get_distortion_center(),

53  }

64  ip.get_lines(), ldm, w,
65  h, vtt);

75  ip.get_lines(), ldm, w, h,
76  vtt);
77  tt_ldm = ldm;
78  }
79 
80  if((fabs(tf_ldm.get_distortion_center().x-
81  tt_ldm.get_distortion_center().x)>0.2*w) ||
82  (fabs(tf_ldm.get_distortion_center().y-
83  tt_ldm.get_distortion_center().y)>0.2*h) ||
84  (check_invertibility(tt_ldm, w, h) == false)){
85  ldm = tf_ldm;
86  error = tf_error;
87  }

90  ldm = tt_ldm;
91  error = tt_error;
92  }
93  }

100 //Iterative optimization
101 double iterative_optimization(
102  ami::subpixel_image_contours &contours,
103  image_primitives &i_primitives,
104  float distance_point_line_max_hough,
105  int max_lines,
106  float angle_resolution,
107  float distance_resolution,
108  float distortion_parameter_resolution,
109  float angle_point_orientation_max_difference,
110  bool opt_center,
111  int width, int height
112 )
113 {
114  double final_error=0.;
115 
116  //We initiallize the previous model, the best one and the previous set of primitives
117  i_primitives.get_distortion().get_d().resize(3);
118  i_primitives.get_distortion().get_d()[2] = 0.;
119  lens_distortion_model previous_model = i_primitives.get_distortion();
120  lens_distortion_model best_model = previous_model;
121  image_primitives previous_ip = i_primitives;
122  //Tolerance for convergence
123  double TOL = 1e-2;
124  //Number of converngence iterations
125  int convergence_iterations = 0;
126  //Fails counter
127  int fail_count = 0;
128  //Number of points: current, best and next
129  int num_points = count_points(i_primitives);
130  int best_num_points = num_points;
131  int next_num_points = num_points+TOL;
132  lens_distortion_model ldm_tf, ldm_tt;
133  //We apply the process until the number of points is not significantly greater
134  //or until the process fails three times
135  while((next_num_points >= (num_points*(1+TOL))) || (fail_count<3))
136  {
137  double error = energy_minimization(previous_model, i_primitives, width,
138  height, opt_center);
139  if(check_invertibility(previous_model, width, height) == false)
140  {
141  if(fail_count==3)
142  break;
143  }
144 
145  i_primitives.clear();
146  //CALL TO IMPROVED HOUGH WITH THE MODEL COMPUTED BEFORE
147  line_equation_distortion_extraction_improved_hough(
148  contours,i_primitives,distance_point_line_max_hough,
149  max_lines,angle_resolution,
150  distance_resolution, 0., 0.,
151  distortion_parameter_resolution,
152  angle_point_orientation_max_difference,
153  true, previous_model);
154 
155  int local_num_points = count_points(i_primitives);
156  if(local_num_points>next_num_points)
157  {
158  //We update the primitives only if the result is better
159  if(local_num_points>best_num_points)
160  {
161  previous_ip = i_primitives;
162  best_num_points = local_num_points;
163  best_model = previous_model;
164  final_error = error;
165  }
166  }
167  else
168  {
169  fail_count++;
170  if(fail_count==3) //THE LIMIT OF FAILS IS 3
171  break;
172  }
173  num_points = next_num_points;
174  next_num_points = local_num_points;
175  convergence_iterations++;
176  }
177  //We take the last and best image primitives object and model
178  i_primitives = previous_ip;
179  i_primitives.set_distortion(best_model);
180 
181  //We return the average error
182  return (final_error/count_points(i_primitives));
183 }
