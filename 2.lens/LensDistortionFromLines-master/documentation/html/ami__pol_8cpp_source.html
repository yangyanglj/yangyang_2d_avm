27 long double ami_horner(
28  long double *pol ,
30  int degree ,
31  long double x ,
32  long double *fp )
33 {
34  long double PPX=pol[degree],PX=pol[degree];
35  int k;
36  for(k=degree-1; k>0; k--) {
37  PX=PX*x+pol[k];
38  PPX=PPX*x+PX;
39  }
40  PX=PX*x+pol[0];
41  *fp=PPX;
42  return(PX);
43 }

53 long double ami_root_bisection(
54  long double *pol ,
56  int degree ,
57  long double a ,
58  long double b ,
59  long double TOL )
60 {
61  long double a2=a,b2=b,fa,fb,c,c0,fc,fp=0.;
62  int iter=0,maxiter=100000;
63 
64  /* we evaluate the polynomial at interval extrema */
65  fa=ami_horner(pol,degree,a,&fp);
66  fb=ami_horner(pol,degree,b,&fp);
67  /* we check if there is a root in the interval */
68  if(fa*fb>0) return(0.5*(a+b));
69  /* we evaluate the interval middle point */
70  c=(a2+b2)*0.5;
71  c0=c+1e30;
72  /* we perform Newton-Raphson or Bisection iterations */
73  while( (b2-a2)>(TOL*fabs(b2)+1e-10) && fabs(c-c0)>(TOL*fabs(c0)+1e-10) &&
74  (iter++)<maxiter) {
75  fc=ami_horner(pol,degree,c,&fp);
76  /* we try Newton-Raphson */
77  c0=c;
78  if(fp!=0.) {
79  c=c-fc/fp;
80  /* printf("c=%f\n",(double) c); */
81  if( c>=a2 && c<=b2 ) continue;
82  }
83  /* if Newton-Raphson fails, we try bisection */
84  c=(a2+b2)*0.5;
85  fc=ami_horner(pol,degree,c,&fp);
86  c0=c+1e30;
87  /* we check if we have arrive to the root */
88  if(fabs(fc)<1e-20) break;
89  /* we perform bisection iteration */
90  if((fa*fc)<0) {
91  b2=c;
92  fb=fc;
93  }
94  else {
95  a2=c;
96  fa=fc;
97  }
98  c=(a2+b2)*0.5;
99  }
100  if(iter>=maxiter){return((a2+b2)*0.5);}
101  return(c);
102 }

113 int ami_polynomial_root(
114  double *pol ,
118  int degree ,
119  double *root_r ,
120  double *root_i )
124 {
125 
126  long double a,b,*p,*p_aux,*ap,*f,fp,*pol2,fa,fb,TOL=1e-14;
127  long double max,n_factor;
128  int m,k,l,Nr,N=degree;
129 
130  p=p_aux=ap=f=pol2=NULL;
131  fp=0.0;
132  a=b=fa=fb=max=n_factor=0.0; /*Added by Luis Gomez*/
133  Nr=0;/*Added by Luis Gomez*/
134 
135  /* WE ALLOCATE MEMORY */
136  p=(long double*)malloc(sizeof(long double)*(N+2));
137  p_aux=(long double*)malloc(sizeof(long double)*(N+2));
138  ap=(long double*)malloc(sizeof(long double)*(N+1));
139  f=(long double*)malloc(sizeof(long double)*(N+1));
140  pol2=(long double*)malloc(sizeof(long double)*(N+2));
141 
142  /* POLYNOMIAL NORMALIZATION */
143  if(pol[0]!=1.) {
144  for(k=0; k<=N; k++) pol[k]/=pol[0];
145  }
146 
147  /* WE REORDER THE POLYNOM */
148  for(k=0; k<=N; k++) pol2[k]=pol[N-k];
149  pol2[N+1]=0.0; /*Added by Luis Gomez*/
150 
151  /* WE NORMALIZE POLINOMIAL COEFFICIENTS TO AVOID POLYNOMIAL EVALUATION IN
152  LARGE NUMBERS*/
153  n_factor=0;
154  for(k=0; k<N; k++) {
155  if(fabs(pol2[k])>10.) {
156  max=log(fabs((double) pol2[k])/10.)/(N-k);
157  if(max>n_factor) n_factor=max;
158  }
159  }
160 
161  n_factor=exp((double) n_factor);
162  max=n_factor;
163  for(k=N-1; k>=0; k--) {
164  pol2[k]/=max;
165  max*=n_factor;
166  }
167 
168  /* WE COMPUTE FACTORIAL NUMBERS */
169  f[0]=1.;
170  for(k=1; k<=N; k++) f[k]=f[k-1]*k;
171 
172  /* WE COMPUTE THE INITIAL INTERVAL WHERE ALL ROOTS ARE INCLUDED */
173  max=fabs(pol2[0]);
174  for(k=1; k<N; k++) {
175  if(fabs(pol2[k])>max)
176  max=fabs(pol2[k]);
177  }
178  /* WE INITIALIZE THE INTERVALS */
179  p[0]=-1.-max/fabs(pol2[N]);
180  /* WE COMPUTE THE POLYNOMIAL DEGREE-1 DERIVATE ROOT */
181  p[1]=-(pol2[N-1]*f[N-1])/(pol2[N]*f[N]);
182  for(k=2; k<=(N+1); k++) p[k]=-p[0];
183  for(k=0; k<=N; k++)p_aux[k]=p[k];
184  p_aux[N+1]=0.0; /*Added by Luis Gomez*/
185  /* WE COMPUTE THE DERIVATIVE POLINOMIAL ROOTS IN A RECURSIVE WAY */
186  for(k=N-2; k>=0; k--) {
187  /* we compute polynomial derivative coefficients */
188  for(l=0; l<=(N-k); l++) {
189  ap[l]=pol2[l+k]*(f[k+l]/f[l]);
190  }
191  /*we check the valid intervals to compute derivative polynomial roots*/
192  m=1;
193  fa=ami_horner(ap,N-k,p_aux[0],&fp);
194  for(l=1; l<=(N-k); l++) {
195  fb=ami_horner(ap,N-k,p_aux[l],&fp);
196  if((fa*fb)<=0 || fabs(fb)<=TOL) {
197  p[m++]=p_aux[l];
198  fa=fb;
199  }
200  }
201  for(l=m; l<N; l++) p[l]=p[N];
202  /* we compute the derivative polynomial roots in each interval */
203  for(l=1; l<=(N-k); l++) {
204  p_aux[l]=ami_root_bisection(ap,N-k,p[l-1],p[l],TOL);
205  }
206  }
207 
208  /* WE STORE THE ROOTS */
209  root_i[0]=0.; /* we fit the complex component of the root to 0 */
210  Nr=0;
211  for(k=1; k<=N; k++) {
212  /* we check if the polynomial has a root in the point */
213  a=(p_aux[k]+p_aux[k-1])*0.5;
214  b=(p_aux[k]+p_aux[k+1])*0.5;
215  fa=ami_horner(pol2,degree,a,&fp);
216  fb=ami_horner(pol2,degree,b,&fp);
217  if(fa*fb<0 || fabs(ami_horner(pol2,degree,p_aux[k],&fp))<TOL) {
218  root_i[Nr]=0; /* we fit the complex component of the root to 0 */
219  root_r[Nr++]=p_aux[k]*n_factor; /* we denormalize the root */
220  }
221  }
222  /* we free the memory */
223  free(p);
224  free(ap);
225  free(f);
226  free(pol2);
227  free(p_aux);
228  /* we return the number of real roots estimated */
229  return(Nr);
230 
231 }
