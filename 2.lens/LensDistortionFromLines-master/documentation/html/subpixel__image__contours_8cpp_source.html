151 void subpixel_image_contours::clean(
152 const int neighborhood_radius,
153 const int min_neighbor_points,
154 const double min_orientation_value,
155 const int min_distance_point)
156 {
157  // WE CHECK THAT THE CONTOURS ARE NOT EMPTY
158  if(c==NULL || x==NULL || y==NULL || coseno==NULL || seno==NULL) return;
159 
160  // WE CHECK PARAMETER neighborhood_radius
161  if(neighborhood_radius<=0) return;
162 
163  int Nedges=index.size();
164  if(Nedges==0) return;
165 
166  // AUXILIARY VARIABLES
167  vector<double> scalar_product(width*height,0.);
168  vector<int> number_neighborhood_points(width*height,0);
169  int i_min=neighborhood_radius;
170  int i_max=width-neighborhood_radius;
171  int j_min=neighborhood_radius;
172  int j_max=height-neighborhood_radius;
173 
174  // WE COMPUTE THE NUMBER OF NEIGHBORHOOD POINTS,
175  // THE NEIGBORHOOD SCALAR PRODUCT AND WE FILL index VECTOR
176  int p=0;
177  for(int j=j_min;j<j_max;j++){
178  int mj=j*width;
179  for(int i=i_min;i<i_max;i++){
180  int m=mj+i;
181  if(c[m]==0) continue;
182  index[p++]=m;
183  for(int k=-neighborhood_radius;k<=neighborhood_radius;k++){
184  int k2=m+k*width;
185  for(int l=-neighborhood_radius;l<=neighborhood_radius;l++){
186  int n=k2+l;
187  if(c[n]==0 || n==m) continue;
188  scalar_product[m]+=coseno[n]*coseno[m]+seno[n]*seno[m];
189  number_neighborhood_points[m]++;
190  }
191  }
192  }
193  }
194  // WE RESIZE index VECTOR
195  index.resize(p);
196 
197  // WE REMOVE CONTOUR POINTS ACCORDING TO THE NUMBER OF NEIGHBORHOOD POINTS
198  // AND STABILITY OF POINT ORIENTATION (CORNERS ARE REMOVED)
199  for(int Np=index.size(),q=0;q<Np;q++){
200  int m=index[q];
201  if(number_neighborhood_points[m]<min_neighbor_points){
202  c[m]=0;
203  continue;
204  }
205  if(scalar_product[m]<(number_neighborhood_points[m]*min_orientation_value)){
206  c[m]=0;
207  continue;
208  }
209  }
210 
211  // WE REMOVE ISOLATED POINTS USING AND ITERATIVE PROCEDURE
212  for(int s=0;s<4;s++){
213  // WE COMPUTE THE NUMBER OF POINTS IN THE NEIGHBORHOOD
214  for(int Np=index.size(),q=0;q<Np;q++){
215  int m=index[q];
216  if(c[m]==0) continue;
217  number_neighborhood_points[m]=0;
218  for(int k=-neighborhood_radius;k<=neighborhood_radius;k++){
219  int k2=m+k*width;
220  for(int l=-neighborhood_radius;l<=neighborhood_radius;l++){
221  int n=k2+l;
222  if(c[n]==0 || n==m) continue;
223  number_neighborhood_points[m]++;
224  }
225  }
226  }
227  // WE REMOVE ISOLATED CONTOUR POINTS
228  for(int Np=index.size(),q=0;q<Np;q++){
229  int m=index[q];
230  if(c[m]==0) continue;
231  if(number_neighborhood_points[m]<neighborhood_radius){
232  c[m]=0;
233  }
234  }
235  }
236 
237  // WE KEEP ONLY 1 CONTOUR POINT EN EACH WINDOW OF RADIUS min_distance_point
238  if(min_distance_point>0){
239  i_min=min_distance_point;
240  i_max=width-min_distance_point;
241  j_min=min_distance_point;
242  j_max=height-min_distance_point;
243  int window_size=2*min_distance_point+1;
244  for(int j=j_min;j<j_max;j+=window_size){
245  int mj=j*width;
246  for(int i=i_min;i<i_max;i+=window_size){
247  int m=mj+i;
248  double max=scalar_product[m];
249  int m_max=m;
250  for(int k=-min_distance_point;k<=min_distance_point;k++){
251  int k2=m+k*width;
252  for(int l=-min_distance_point;l<=min_distance_point;l++){
253  int n=k2+l;
254  if(c[n]==0 || m==n) continue;
255  if(scalar_product[n]>max){
256  max=scalar_product[n];
257  c[m_max]=0;
258  m_max=n;
259  }
260  else { //if(scalar_product[n]<max){
261  c[n]=0;
262  }
263  }
264  }
265  }
266  }
267  }
268 
269  // WE FILL INDEX VECTOR
270  int m=0;
271  for(int Np=index.size(),q=0;q<Np;q++){
272  if(c[index[q]]>0) index[m++]=index[q];
273  }
274  index.resize(m);
275  return;
276 }
