501 void compute_ps(double &p1, double &p2, lens_distortion_model &ldm,
502  int w, int h, bool quo)
503 {
504  double dmi = update_rsqmax(ldm.get_distortion_center(),w,h);
505  double r1 = sqrt(dmi);
506  double r2 = r1/2;
507  double k1,k2;
508  if(ldm.get_d().size()<3)
509  {
510  k1 = ldm.get_d()[1];
511  k2 = 0.;
512  }
513  else
514  {
515  k1 = ldm.get_d()[1];
516  k2 = ldm.get_d()[2];
517  }
518  if(quo)
519  {
520  p1 = (1/(1+k1*r1*r1+k2*r1*r1*r1*r1))-1;
521  p2 = (1/(1+k1*r2*r2+k2*r2*r2*r2*r2))-1;
522  }
523  else
524  {
525  p1 = k1*4*r2*r2 + k2*16*r2*r2*r2*r2;
526  p2 = k1*r2*r2 + k2*r2*r2*r2*r2;
527  }
528 }
