366 void canny(ami::image<T> input ,
367  ami::image<T> &output ,
368  float *seno ,
369  float *coseno ,
370  int *x ,
371  int *y ,
372  float per_low ,
373  float per_high )
374 {
375  //cout << "AMI::FILTERS::CANNY STARTS" << endl;
376  bool neighborhoodtype = 9; //9 NEIGHBOURS FOR THE GRADIENT
377  int width = input.width();
378  int height = input.height();
379  int size = width*height;
380  //THRESHOLDS
381  float low_threshold,high_threshold;
382  //MAP
383  ami::image<int> imap(width,height,1,0);
384  //NORM OF THE GRADIENT
385  float *gradient_norm = new float[size];
386  //GRADIENT COMPONENTS
387  ami::image<float> x_grad(width,height,input.nChannels(),0);
388  ami::image<float> y_grad(width,height,input.nChannels(),0);
389  //IMAGE FOR THE RESULT OF THE CONVOLUTION WITH THE GAUSSIAN
390  ami::image<float> blurred(width,height,input.nChannels(),0);
391 
392  //WE APPLY A GAUSSIAN CONVOLUTION TO THE INPUT IMAGE (NOISE REDUCTION)
393  gauss_conv(input,blurred,2.,2.,2.);
394 
395  //WE COMPUTE THE GRADIENT OF THE BLURRED IMAGE
396  grad(blurred,x_grad,y_grad,neighborhoodtype);
397  for(int k=0; k<size; k++)
398  gradient_norm[k] = sqrt((x_grad[k]*x_grad[k]) + (y_grad[k]*y_grad[k]));
399 
400  //WE COMPUTE THE THRESHOLDS
401  low_threshold = ami_median_float(per_low*size, size, gradient_norm);
402  high_threshold = ami_median_float(per_high*size, size, gradient_norm);
403  //WE IMPOSE A MINIMUM VALUE FOR THE THRESHOLDS
404  low_threshold=(low_threshold<2)?2:low_threshold;
405  high_threshold=(high_threshold<4)?4:high_threshold;
406 
407 
408  for(int i=0; i<height; i++)
409  {
410  for(int j=0; j<width; j++)
411  {
412  int pos = i*width+j;
413  //WE PUT THE MAP = 2 FOR VALUES > HIGH_THRESHOLD
414  if(gradient_norm[pos] > high_threshold)
415  {
416  imap[pos] = 2;
417  }
418  //LIMITS OF THE IMAGE
419  if((i==0) || (i==height-1) || (j==0) || (j==width-1))
420  {
421  imap[pos] = 1;
422  }
423  }
424  }
425 
426  //NON-MAXIMUM SUPPRESSION
427  float c = 1/(sqrt(2.));
428  for(int i=1; i<height-1; i++)
429  {
430  for(int j=1; j<width-1; j++)
431  {
432  int k = i*width+j;
433  //ORIENTATIONS
434  float o1 = fabs(x_grad[k]);
435  float o2 = fabs(y_grad[k]);
436  float o3 = fabs(c*(x_grad[k]+y_grad[k]));
437  float o4 = fabs(c*(x_grad[k]-y_grad[k]));
438  //NEIGHBOURS
439  float neigh1=0;
440  float neigh2=0;
441  //WE SELECT THE MAXIMUM OF THE ORIENTATIONS
442  float candidate = max(o1,max(o2,max(o3,o4)));
443  //WE TAKE THE VALUE OF THE NEIGHBOURS BASED ON MAXIMUM ORIENTATION
444  //0º
445  if(candidate == o1)
446  {
447  //O - E
448  neigh1 = gradient_norm[k-1];
449  neigh2 = gradient_norm[k+1];
450  }
451  //90º
452  if(candidate == o2)
453  {
454  //N - S
455  neigh1 = gradient_norm[k-width];
456  neigh2 = gradient_norm[k+width];
457  }
458  //45º
459  if(candidate == o3)
460  {
461  //NE - SO
462  neigh1 = gradient_norm[k-width+1];
463  neigh2 = gradient_norm[k+width-1];
464  }
465  //135º
466  if(candidate == o4)
467  {
468  //NO - SE
469  neigh1 = gradient_norm[k-width-1];
470  neigh2 = gradient_norm[k+width+1];
471  }
472  //WE UPDATE THE RESULT IN FUNCTION OF THE MAXIMUM VALUE OF
473  //THE GRADIENT MODULE BETWEEN THE POINT AND ITS NEIGHBOURS
474  float p_grad = gradient_norm[k]; //GRADIENT VALUE IN THE POINT
475  if((p_grad>neigh1) && (p_grad>neigh2))
476  {
477  if(gradient_norm[k] >= high_threshold)
478  imap[k] = 2;
479  else
480  imap[k] = 0;
481  }
482  else
483  {
484  imap[k] = 1;
485  }
486  }
487  }
488 
489  //HISTERESIS PROCESS OF THE CANNY METHOD
490  for(int k=0; k<size; k++)
491  {
492  if(imap[k]==2)
493  {
494  //NEIGHBOURS = O, E, N, S, NO, NE, SO, SE
495  int nv[8] = {k-1, k+1, k-width, k+width,
496  k-width-1, k-width+1, k+width-1, k+width+1};
497  for(int iv=0; iv<8; iv++)
498  {
499  int nk = nv[iv];
500  if(imap[nk]>1) continue;
501  if((imap[nk]==0) && (gradient_norm[nk]>low_threshold))
502  {
503  fill_imap(imap, gradient_norm, low_threshold, nk);
504  }
505  else
506  imap[nk] = 1;
507  }
508  }
509  }
510 
511  /*float maxi=0.;
512  for(int i=0; i<size; i++)
513  {
514  maxi = max(maxi, gradient_norm[i]);
515  }*/
516 
517  //FILL THE OUTPUT IMAGE TAKING INTO ACCOUNT THE IMAP IMAGE VALUES
518  //THE EDGES ARE THE PIXELS WITH MAP > 1
519  for(int i=1; i<height-1; i++)
520  {
521  for(int j=1; j<width-1; j++)
522  {
523  int pos = i*width+j;
524  if(imap[pos]>1)
525  {
526  output[pos] = 255;
527  //VALUE OF THE EDGES (NORMALIZED TO THE RANGE OF 0..255)
528  /*output[pos] = (unsigned char)(255*
529  gradient_norm[pos]/maxi);

531  gradient_norm[pos]/maxi);

534

538

541  y[pos] = i;

549  }
